def plotXPoints(X,x0):
    fig,ax = plt.subplots(1,1)
    p_t = t[:(x0+X)][x0:]
    p_bkgd = bkgd[:(x0+X)][x0:]
    plt.plot(p_t,p_bkgd,'o')
    plt.title("Scatter Plot of " + str(X) + " Points starting from Point #" + str(x0),fontsize=16)
    plt.show()
